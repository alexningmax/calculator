<script>
const g=9.8,rho=1.225,Cd=0.5,d=0.04,S=Math.PI*(d/2)**2,m=0.0027;
const bounceRatio=0.55,dt=0.0005,RULE_LEN=40;
let fireH=0.32,fireV0=0;
const line=document.getElementById('angleLine');
const dot=document.getElementById('angleDot');
const showAngle=document.getElementById('showAngle');
const angleInput=document.getElementById('angleInput');
let currentAngle=0,isDrag=false;

function setDiskAngle(deg){
deg = ((deg % 360) + 360) % 360;
currentAngle = deg;
line.style.transform = `rotate(${deg - 90}deg)`;
dot.style.transform = `rotate(${deg - 90}deg) translateX(100px)`;
showAngle.innerText = deg.toFixed(1);
angleInput.value = deg.toFixed(1);
}

function getAngleFromPos(x,y){
let rect = document.querySelector('.angle-disk').getBoundingClientRect();
let cx = rect.left + rect.width/2;
let cy = rect.top + rect.height/2;
let dx = x - cx;
let dy = cy - y;
let rad = Math.atan2(dx, dy);
let deg = rad * 180 / Math.PI;
if(deg < 0) deg += 360;
return deg;
}

dot.addEventListener('mousedown', e => { e.preventDefault(); isDrag = true; });
document.addEventListener('mouseup', () => isDrag = false);
document.addEventListener('mousemove', e => {
if(!isDrag) return;
e.preventDefault();
setDiskAngle(getAngleFromPos(e.clientX, e.clientY));
});

dot.addEventListener('touchstart', e => { e.preventDefault(); isDrag = true; }, {passive: false});
document.addEventListener('touchend', () => isDrag = false);
document.addEventListener('touchmove', e => {
if(!isDrag) return;
e.preventDefault();
let t = e.touches[0];
setDiskAngle(getAngleFromPos(t.clientX, t.clientY));
}, {passive: false});

angleInput.addEventListener('input', () => {
let val = parseFloat(angleInput.value);
if(!isNaN(val)) setDiskAngle(val);
});

// 模拟函数：返回水平位移和落地时的垂直速度
function simulate(compassAngle, hStart, v0){
let emitAngle = 90 - compassAngle;
let rad = Math.PI / 180 * emitAngle;
let vx = v0 * Math.cos(rad);
let vy = v0 * Math.sin(rad);
let x = 0, y = hStart;
while(y > 0){
let v = Math.hypot(vx, vy);
let drag = 0.5 * rho * Cd * S * v * v;
let ax = -drag * vx / m;
let ay = -g - drag * vy / m;
vx += ax * dt;
vy += ay * dt;
x += vx * dt;
y += vy * dt;
}
return {disX: x, endVy: vy};
}

function calibrate(baseCm){
let target = baseCm / 100;
let lowV = 1.0, highV = 5.0;
for(let i=0;i<60;i++){
let midV = (lowV + highV) / 2;
simulate(90, fireH, midV).disX < target ? lowV = midV : highV = midV;
}
fireV0 = (lowV + highV) / 2;
}

function calcRule(n, cutVal){
let realVal = Math.max(-9999, parseFloat(n) - parseFloat(cutVal));
return {
count: Math.floor(realVal / RULE_LEN),
last: (realVal % RULE_LEN).toFixed(1)
};
}

function calc(){
let baseCm = parseFloat(document.getElementById("baseLen").value);
let laserCut = parseFloat(document.getElementById("laserCut").value) || 0;
let ang = currentAngle;
let res = document.getElementById("res");
let ruleTip = document.getElementById("ruleTip");

if(isNaN(baseCm) || baseCm <= 0){
res.innerText = "请输入合法基准射程";
ruleTip.innerText = "";
return;
}
if(laserCut < 0){
res.innerText = "减去长度不能为负数";
ruleTip.innerText = "";
return;
}

calibrate(baseCm);

// 第一段：按罗盘角度发射
let p1 = simulate(ang, fireH, fireV0);
let firstLandX = p1.disX * 100;       // 转换为厘米
let r1 = firstLandX.toFixed(2);

// 第二段：从第一个落点弹跳，水平发射，弹跳高度由垂直速度反弹决定
let bounceH2 = Math.abs(p1.endVy) * bounceRatio / g;
let p2 = simulate(90, bounceH2, fireV0);
let secondLandX = firstLandX + p2.disX * 100;
let r2 = secondLandX.toFixed(2);

// 第三段：从第二个落点弹跳
let bounceH3 = Math.abs(p2.endVy) * bounceRatio / g;
let p3 = simulate(90, bounceH3, fireV0);
let thirdLandX = secondLandX + p3.disX * 100;
let r3 = thirdLandX.toFixed(2);

// 减去激光误差
let realR1 = Math.max(-9999, parseFloat(r1) - laserCut).toFixed(2);
let realR2 = Math.max(-9999, parseFloat(r2) - laserCut).toFixed(2);
let realR3 = Math.max(-9999, parseFloat(r3) - laserCut).toFixed(2);

let g1 = calcRule(r1, laserCut);
let g2 = calcRule(r2, laserCut);
let g3 = calcRule(r3, laserCut);

res.innerHTML = `基准射程：${baseCm} cm（90°水平）<br>发射仰角：${ang.toFixed(1)} °<br>激光减去：${laserCut.toFixed(1)} cm<br><br>第一段（初始发射）：${r1} cm<br>第二段（第一次弹跳）：${r2} cm<br>第三段（第二次弹跳）：${r3} cm<br><br>减去激光误差后：<br>第一段：${realR1} cm<br>第二段：${realR2} cm<br>第三段：${realR3} cm`;

ruleTip.innerHTML = `📏 40cm尺子参考（减后）：<br>第一段：${g1.count}根整尺，剩余 ${g1.last} cm<br>第二段：${g2.count}根整尺，剩余 ${g2.last} cm<br>第三段：${g3.count}根整尺，剩余 ${g3.last} cm`;
}

// 初始化为水平发射（90°）// 初始化为水平发射（90°）
setDiskAngle(90);
calc();
</script>
